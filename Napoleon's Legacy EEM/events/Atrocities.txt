			limit = { NOT = { has_global_flag = nukes_used_global } }
			owner = { set_global_flag = nukes_used_global }
			owner = { set_country_flag = first_nuclear_bombing_campaign }
		}

		add_country_modifier = { name = nukes_used_recently duration = 365 }
		badboy = 20
		any_pop = { limit = { NOT = { type = clergymen } } dominant_issue = { value = pacifism factor = 0.1 } }
